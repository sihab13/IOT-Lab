{

BLYNK_WRITE(V0) // This function will be called every time the virtual button V0 is pressed
{
  int value = param.asInt(); // Get the value of the virtual button V0
  if (value == 1) {
    digitalWrite(ledPin, HIGH); // Turn on the LED
  } else {
    digitalWrite(ledPin, LOW); // Turn off the LED
  }
}
