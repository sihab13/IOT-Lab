#define BLYNK_TEMPLATE_NAME "Manhole Monitoring System"
#define BLYNK_AUTH_TOKEN "sOOSJKuS-eUZN5MZVpJk5wmVuXNNyQPJ"

#include <ESP8266WiFi.h>
#include <BlynkSimpleEsp8266.h>
#include <DHT.h>
#include <NewPing.h>

char ssid[] = "Bachelor";
char pass[] = "hp515.com";

#define DHTPIN D1     // DHT11 data pin is connected to NodeMCU D0
#define DHTTYPE DHT11 // DHT11 sensor type

#define TRIGGER_PIN D2
#define ECHO_PIN D3
#define MAX_DISTANCE 8
#define MAX_DEPTH 8

DHT dht(DHTPIN, DHTTYPE);
NewPing sonar(TRIGGER_PIN, ECHO_PIN, MAX_DISTANCE);

void setup()
{
  Serial.begin(115200);
  Blynk.begin(BLYNK_AUTH_TOKEN, ssid, pass);

  dht.begin();
}

void loop()
{
  float temperature = dht.readTemperature();
  float humidity = dht.readHumidity();

  if (isnan(temperature) || isnan(humidity)) {
    Serial.println("Failed to read from DHT sensor!");
    return;
  }

  Serial.print("Temperature: ");
  Serial.print(temperature);
  Serial.print(" Â°C, Humidity: ");
  Serial.print(humidity);
  Serial.println(" %");

  Blynk.virtualWrite(V0, temperature); // Send temperature value to Blynk app
  Blynk.virtualWrite(V1, humidity);    // Send humidity value to Blynk app

  unsigned int distance = sonar.ping_cm();
  Blynk.virtualWrite(V2, distance); // Send distance value to Blynk app

  Serial.print("Distance: ");
  Serial.print(distance);
  Serial.print(" cm");

  float water_depth = MAX_DEPTH - distance;
  Serial.print(", Water Depth: ");
  Serial.print(water_depth);
  Serial.print(" cm");

  float water_level_percentage = (water_depth / MAX_DEPTH) * 100;
  Blynk.virtualWrite(V3, water_level_percentage); // Send water level percentage to Blynk app

  Serial.print(", Water Level Percentage: ");
  Serial.print(water_level_percentage);
  Serial.println(" %");

  Blynk.run();
  delay(2000); // Wait for 2 seconds before taking next reading
}
